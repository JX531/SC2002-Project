StudentsList
Name - Email - Faculty

StaffList
Name - Email - Faculty

Camp_Student
Relation - CampName- StudentID - Points
where relation is 0, 1 or 2
0 = withdrawn
1 = attendee
2 = committee
Matching Students and the camps they are withdrawn/participating in & points, default 0

Camp_Staff
CampName - StaffID
Matching staff and their created camps

EnquiryList
SentTo - MadeBy - Question - Answer
where SentTo is CampID and MadeBy is Student ID

SuggestionList
SentTo - MadeBy - Suggestion - Approved - Processed
where SentTo is CampID and MadeBy is Student ID
